//1672. Richest Customer Wealth
class Solution {
    public int maximumWealth(int[][] accounts) {
        int maxWealth = 0;
        for (int[] account : accounts) {
            int accountSum = 0;
            for (int i = 0; i < account.length; i++) {
                accountSum += account[i];
            }

            if (accountSum > maxWealth) {
                maxWealth = accountSum;
            }
        }
        return maxWealth;
    }
}
