//283. Move Zeroes
class Solution {
    public void moveZeroes(int[] nums) {
        int n=nums.length,pos=0;
        for(int i=0;i<n;i++){
            if(nums[i]!=0){
                nums[pos++]=nums[i];
            }
        }
        for(int i=pos;i<n;i++){
            nums[i]=0;
        }
    }
}
