//2455. Average Value of Even Numbers That Are Divisible by Three
class Solution {
    public int averageValue(int[] nums) {
        int sum=0;
        int i=0;
        for(int n:nums){
            if(n%6==0){
                sum+=n;
                i++;
            }
        }
        if (i==0) return i;
        return sum/i;
    }
}
