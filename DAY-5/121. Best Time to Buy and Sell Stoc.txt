//121. Best Time to Buy and Sell Stock
class Solution {
    public int maxProfit(int[] prices) {
        int buy=prices[0],sell=0;
        int profit=0;
        for(int i=0;i<prices.length;i++){
            if(buy>prices[i]){
                buy=prices[i];
            }
            if(profit<prices[i]-buy){
                profit=prices[i]-buy;
            }
        }
        return profit;
    }
}
